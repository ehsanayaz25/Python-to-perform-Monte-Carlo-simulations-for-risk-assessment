import numpy as np
import matplotlib.pyplot as plt

# Constants
p = 1  # Replace with the value of p if known

# Define the equations
def equation1(x, p):
    return 4 * x**3 + 2 * x**2 + 10 * (50 * p)

def equation2(x, p):
    return np.log(x) - (50 * p)

def equation3(x, p):
    return np.exp(x) * (50 * p)

# Generate x values
x = np.linspace(0.1, 10, 500)  # Avoid x = 0 due to log(x)

# Compute y values
y1 = equation1(x, p)
y2 = equation2(x, p)
y3 = equation3(x, p)

# Plot the equations
plt.figure(figsize=(12, 6))
plt.plot(x, y1, label='y = 4x^3 + 2x^2 + 10(50p)')
plt.plot(x, y2, label='y = log(x) - (50p)')
plt.plot(x, y3, label='y = e^x(50p)')
plt.axhline(0, color='black', linewidth=0.5, linestyle='--')
plt.axvline(0, color='black', linewidth=0.5, linestyle='--')
plt.xlabel('x')
plt.ylabel('y')
plt.title('Graphs of Given Equations')
plt.legend()
plt.grid(True)
plt.show()
